```python
def smoothstep(t: float) -> float:
    """Hermite smoothstep for S-curve interpolation.
    Returns 0 for t<=0, 1 for t>=1, smooth transition between.
    Formula: 3t^2 - 2t^3
    """
    t = max(0.0, min(1.0, t))
    return t * t * (3.0 - 2.0 * t)
```
