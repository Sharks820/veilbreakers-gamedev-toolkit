```python
veg_counts[veg_key] = veg_counts.get(veg_key, 0) + 1

# Try the combined key first, then just the type
gen_entry = VEGETATION_GENERATOR_MAP.get(veg_key) or VEGETATION_GENERATOR_MAP.get(p['type'])
if gen_entry:
    gen_func, gen_kwargs = gen_entry
    mesh_spec = gen_func(**gen_kwargs, seed=seed + veg_counts[veg_key])
    if isinstance(mesh_spec, dict) and mesh_spec.get("vertices"):
        instance = mesh_from_spec(mesh_spec, name=f"{veg_key}_{veg_counts[veg_key]:04d}")
    else:
        # Generator returned but no geometry -- fall back to small cube with warning
        import logging
        logging.getLogger(__name__).warning("Vegetation generator %s returned empty mesh, using fallback", veg_key)
        placeholder_mesh = bpy.data.meshes.new(f"_veg_{veg_key}_{veg_counts[veg_key]:04d}")
        instance_bm = bmesh.new()
        bmesh.ops.create_cube(instance_bm, size=0.5)
        instance_bm.to_mesh(placeholder_mesh)
        instance_bm.free()
        instance = bpy.data.objects.new(f"{veg_key}_{veg_counts[veg_key]:04d}", placeholder_mesh)
else:
    # No generator found -- fall back to small cube with warning
    import logging
    logging.getLogger(__name__).warning("No VEGETATION_GENERATOR_MAP entry for %s, using cube fallback", veg_key)
    placeholder_mesh = bpy.data.meshes.new(f"_veg_{veg_key}_{veg_counts[veg_key]:04d}")
    instance_bm = bmesh.new()
    bmesh.ops.create_cube(instance_bm, size=0.5)
    instance_bm.to_mesh(placeholder_mesh)
    instance_bm.free()
    instance = bpy.data.objects.new(f"{veg_key}_{veg_counts[veg_key]:04d}", placeholder_mesh)
```
