    **Step 3: Opus verification scan (TEST-04).**
    Perform a systematic scan of the generated scene:

    ```
    blender_execute(code="""
    import bpy

    print("=== OPUS VERIFICATION SCAN ===")
    print()

    # 1. Object census
    all_mesh = [o for o in bpy.data.objects if o.type == 'MESH']
    print(f"Total mesh objects: {len(all_mesh)}")

    # 2. Material check -- any objects with no materials?
    no_mat = [o for o in all_mesh if not o.data.materials or all(m is None for m in o.data.materials)]
    print(f"Objects with NO materials: {len(no_mat)}")
    if no_mat:
        for o in no_mat[:10]:
            print(f"  {o.name}")
        if len(no_mat) > 10:
            print(f"  ... and {len(no_mat)-10} more")

    # 3. Z=0 check
    z0 = [o for o in all_mesh if abs(o.location.z) < 0.1 and 'terrain' not in o.name.lower() and 'water' not in o.name.lower()]
    print(f"Objects at Z=0 (suspect): {len(z0)}")
    if z0:
        for o in z0[:10]:
            print(f"  {o.name}: Z={o.location.z:.2f}")

    # 4. Collection structure
    print()
    print("Collection structure:")
    for col in bpy.data.collections:
        objs = len(col.objects)
        if objs > 0:
            print(f"  {col.name}: {objs} objects")

    # 5. Terrain stats
    terrain = [o for o in all_mesh if 'terrain' in o.name.lower()]
    for t in terrain:
        mesh = t.data
        z_vals = [v.co.z for v in mesh.vertices]
        print(f"Terrain '{t.name}': {len(mesh.vertices)} verts, Z={min(z_vals):.1f} to {max(z_vals):.1f}")

    # 6. Building count
    buildings = [o for o in all_mesh if any(kw in o.name.lower() for kw in ['building', 'house', 'shop', 'tavern', 'chapel'])]
    print(f"Building objects: {len(buildings)}")

    # 7. Castle piece count
    castle = [o for o in all_mesh if 'castle' in o.name.lower() or 'keep' in o.name.lower() or 'tower' in o.name.lower()]
    print(f"Castle objects: {len(castle)}")

    # 8. Vegetation count
    veg = [o for o in all_mesh if any(kw in o.name.lower() for kw in ['tree', 'bush', 'grass', 'shrub', 'fern'])]
    print(f"Vegetation objects: {len(veg)}")

    # 9. Interior count
    interior = [o for o in all_mesh if 'interior' in o.name.lower() or any(kw in o.name.lower() for kw in ['tavern_hall', 'forge', 'nave', 'great_hall', 'armory'])]
    print(f"Interior objects: {len(interior)}")

    # 10. Tripo props
    tripo = [o for o in all_mesh if 'tripo' in o.name.lower() or any(kw in o.name.lower() for kw in ['sign', 'well', 'banner', 'altar', 'anvil_stump'])]
    print(f"Tripo/hero props: {len(tripo)}")

    # VERDICT
    print()
    print("=== SCAN VERDICT ===")
    issues = []
    if len(no_mat) > 10:
        issues.append(f"CRITICAL: {len(no_mat)} objects have no materials")
    if len(z0) > 5:
        issues.append(f"WARNING: {len(z0)} objects at Z=0")
    if len(buildings) < 10:
        issues.append(f"WARNING: Only {len(buildings)} buildings (expected 25+)")
    if len(castle) < 5:
        issues.append(f"WARNING: Only {len(castle)} castle pieces (expected 10+)")
    if len(veg) < 50:
        issues.append(f"WARNING: Only {len(veg)} vegetation objects (expected 100+)")
    if len(interior) < 5:
        issues.append(f"WARNING: Only {len(interior)} interior objects (expected 10+)")

    if issues:
        print(f"ISSUES FOUND: {len(issues)}")
        for i in issues:
            print(f"  {i}")
    else:
        print("CLEAN -- All checks passed")
    """)
    ```
