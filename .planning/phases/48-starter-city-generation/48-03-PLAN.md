    ```
    blender_execute(code="""
    import bpy
    water_objects = [o for o in bpy.data.objects if 'water' in o.name.lower()]
    veg_objects = [o for o in bpy.data.objects if any(kw in o.name.lower() for kw in ['tree', 'bush', 'grass', 'shrub', 'fern'])]
    print(f"Vegetation objects: {len(veg_objects)}")
    in_water = 0
    for v in veg_objects:
        for w in water_objects:
            if w.type == 'MESH':
                wb = w.bound_box
                wmin = min(p[0] for p in wb), min(p[1] for p in wb)
                wmax = max(p[0] for p in wb), max(p[1] for p in wb)
                vx, vy = v.location.x, v.location.y
                if wmin[0] <= vx <= wmax[0] and wmin[1] <= vy <= wmax[1]:
                    in_water += 1
                    break
    print(f"Vegetation in water bounds: {in_water}")
    if in_water > 10:
    ```
