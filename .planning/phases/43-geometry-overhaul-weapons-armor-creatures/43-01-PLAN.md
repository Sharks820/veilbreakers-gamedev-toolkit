    ```python
    def _wrap_creature_part_result(
        result: tuple | dict,
        name: str,
        category: str = "creature_part",
    ) -> dict:
        """Wrap creature part generator (verts, faces, groups) tuple into MeshSpec dict."""
        if isinstance(result, dict):
            return result  # Already MeshSpec
        verts, faces, groups = result
        return {
            "vertices": verts,
            "faces": faces,
            "uvs": [],
            "vertex_groups": groups,
            "empties": {},
            "detail_features": [],
            "metadata": {
                "name": name,
                "category": category,
                "vertex_count": len(verts),
                "face_count": len(faces),
                "tri_count": sum(len(f) - 2 for f in faces),
            },
        }
    ```
